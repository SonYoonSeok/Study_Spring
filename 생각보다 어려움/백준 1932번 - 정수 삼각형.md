        int[][] dp = new int[N][N];

        for (int i = 0; i < N; i++) {
            String[] str = sc.nextLine().split(" ");
            for (int j = 0; j < str.length; j++) {
                dp[i][j] = Integer.parseInt(str[j]);
            }
        }

        for (int i = 0; i < N - 1; i++) {
            for (int j = 0; j < i + 2; j++) {
                if (j == 0) {
                    dp[i + 1][j] += dp[i][0];
                } else {
                    dp[i + 1][j] += Math.max(dp[i][j - 1], dp[i][j]);
                }
            }
        }

        int answer = 0;
        for (int i = 0; i < N; i++) {
            if (dp[N - 1][i] > answer) {
                answer = dp[N - 1][i];
            }
        }
        System.out.println(answer);
    }
}
